cmake_minimum_required(VERSION 2.6)

PROJECT(PolyLineParametricPath)

FIND_PACKAGE(ITK REQUIRED)
INCLUDE(${ITK_USE_FILE})

FIND_PACKAGE(VTK REQUIRED)
INCLUDE(${VTK_USE_FILE})

ADD_EXECUTABLE(PolyLineParametricPath PolyLineParametricPath.cpp)
TARGET_LINK_LIBRARIES(PolyLineParametricPath ${ITK_LIBRARIES})

ADD_EXECUTABLE(Output Output.cpp)
TARGET_LINK_LIBRARIES(Output ${ITK_LIBRARIES} ${VTK_LIBRARIES})
